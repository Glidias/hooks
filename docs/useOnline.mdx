Detect whether the network is online.

## Usage

```ts
function useOnline(): boolean;
```

## Example

<ExampleBox>
  <UseOnlineExample />
</ExampleBox>

```tsx
import { useOnline } from "@lilib/hooks";

function Example() {
  const online = useOnline();
  return "The network is: " + (online ? "online" : "offline");
}

export default Example;
```

This hook combines multiple refs into one. It's very useful when developing components.

## API

```ts
function useMergedRef<T>(...refs: Ref<T>[]): React.RefCallback<T>;
```

## Example

```tsx
import React, { forwardRef, useRef } from "react";
import { useMergedRef } from "@lilib/hooks";

const Component = forwardRef<HTMLDivElement, any>((props, ref) => {
  const domRef = useRef<HTMLDivElement>(null);
  const mergedRef = useMergedRef(domRef, ref);

  // Do somthing.

  return <div ref={mergedRef}>...</div>;
});
```

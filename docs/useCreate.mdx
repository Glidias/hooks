This hook is similar to `React.useMemo`, but the deps is an empty array by default, and ensures that the create function is executed only once with deps unchanged.

## API

```ts
function useCreate<T>(create: () => T, deps?: React.DependencyList): T;
```

## Example

```ts
import { useState } from "react";
import { useCreate, useEventListener } from "@lilib/hooks";

const query = "(prefers-color-scheme: dark)";

function useDarkTheme(): boolean {
  const media = useCreate(() => window.matchMedia(query));
  const [dark, setDark] = useState(media.matches);

  useEventListener(media, "change", (event: MediaQueryListEvent) => {
    setDark(event.matches);
  });

  return dark;
}
```

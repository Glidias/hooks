Returns a ref object to track if the component is mounted.

## Usage

```ts
function useMounted(): React.MutableRefObject<boolean>;
```

## Example

```tsx
import React, { useEffect } from "react";
import { useMounted } from "@lilib/hooks";

function Example() {
  const mountedRef = useMounted();
  console.log("Is the component mounted: ", mountedRef.current ? "yes" : "no");
  ...
}
```

Returns a ref object whose `.current` property is assigned to previous updated value.

## Usage

```ts
function usePrevious<T>(value: T): React.MutableRefObject<T>;
```

## Example

```ts
import { useRef } from "react";
import { useMounted, usePrevious } from "@lilib/hooks";

function useTitle(title: string) {
  const mountedRef = useMounted();
  const previousTitleRef = usePrevious(title);

  if (!mountedRef.current || previousTitleRef.current !== title) {
    document.title = title;
  }
}

export default useTitle;
```

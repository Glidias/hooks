Set document title.

## Usage

```ts
function useTitle(title: string, options?: { restore?: boolean }): void;
```

### Params

- **`title`:** document title.
- **`options.restore`:** restore previous title on the component unmount.

## Example

<ExampleBox>
  <UseTitleExample />
</ExampleBox>

```tsx
import React, { useState } from "react";
import { useTitle } from "@lilib/hooks";

function Example() {
  const [title, setTitle] = useState("");
  useTitle(title, { restore: true });

  return (
    <input
      placeholder="Enter title"
      value={title}
      onChange={(event) => {
        setTitle(event.target.value);
      }}
    />
  );
}

export default Example;
```
